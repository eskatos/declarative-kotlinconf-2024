`settings.gradle.dcl`
```kotlin
conventions {
    kotlinJvmLibrary {
        javaVersion = 21
    }
}
```
`build.gradle.dcl`
```kotlin
kotlinJvmLibrary {
    // javaVersion comes from convention

    dependencies {
        api(project(":core:common"))
    }
}
```
